\section*{CONSIDERAÇÕES PRELIMINARES}

\textbf{CONSIDERANDO QUE:}

\begin{enumerate}[label=(\Alph*), itemsep=0.2em]
    \item A Vendedora é titular dos direitos de propriedade intelectual sobre a plataforma de software denominada \textbf{``CRM Solar''}, sistema de gestão de relacionamento com clientes e automação de vendas desenvolvido para o setor de energia solar; e

    \item A Compradora tem interesse em adquirir a totalidade dos direitos de propriedade intelectual sobre o referido software, incluindo código-fonte, documentação, arquitetura e direito de comercialização, para fins de exploração comercial própria no modelo SaaS; e

    \item As Partes desejam estabelecer níveis de serviço (SLA) para o período de suporte inicial e definir responsabilidades claras quanto à proteção de dados pessoais em conformidade com a LGPD.
\end{enumerate}

\noindent\textbf{RESOLVEM AS PARTES}, de mútuo e comum acordo, celebrar este Contrato com as seguintes cláusulas e condições:

\section{DEFINIÇÕES}

\textbf{1.1.} Para os fins deste Contrato:

\begin{itemize}[itemsep=0.1em]
    \item \textbf{Software:} Plataforma CRM Solar, incluindo módulos, código-fonte, documentação e ativos
    \item \textbf{Código-Fonte:} Arquivos de programação em formato editável e legível
    \item \textbf{Propriedade Intelectual:} Direitos autorais, de software, marcas, segredos industriais e know-how
    \item \textbf{SLA:} Acordo de Nível de Serviço com prazos de resposta para suporte técnico
    \item \textbf{Dados Pessoais:} Informação relativa a pessoa natural identificada ou identificável
\end{itemize}

\section{OBJETO}

\textbf{2.1.} O presente Contrato tem por objeto a \textbf{venda e transferência definitiva} à Compradora de todos os direitos de propriedade intelectual sobre o Software ``CRM Solar'', incluindo:

\begin{enumerate}[label=\alph*), itemsep=0.1em]
    \item Código-fonte completo (frontend e backend);
    \item Banco de dados estruturado e migrations;
    \item Documentação técnica e de usuário;
    \item Direito de uso, modificação, reprodução e comercialização;
    \item Direito de sublicenciamento a terceiros;
    \item Arquitetura multi-tenant para operação no modelo SaaS.
\end{enumerate}

\textbf{2.2.} Estão incluídos neste Contrato:

\begin{enumerate}[label=\alph*), itemsep=0.1em]
    \item Suporte técnico durante os primeiros 3 (três) meses após a Entrega;
    \item Manutenção e correção de bugs durante os primeiros 3 (três) meses;
    \item Treinamento de 8 (oito) horas para operação do Software.
\end{enumerate}

\textbf{2.3.} \textbf{NÃO} estão incluídos (responsabilidade da Compradora):

\begin{enumerate}[label=\alph*), itemsep=0.1em]
    \item Custos de infraestrutura (hospedagem, Supabase, Vercel);
    \item Custos de APIs de terceiros (WhatsApp Business, Evolution API);
    \item Desenvolvimento de novas funcionalidades após a Entrega;
    \item Suporte técnico após os 3 (três) primeiros meses.
\end{enumerate}

\section{ENTREGA E ACEITE}

\section{DIREITO DE ARREPENDIMENTO}

\textbf{5.1.} A Compradora poderá se arrepender em até \textbf{14 dias corridos após a Entrega}.

\textbf{5.2.} Em caso de arrependimento:
\begin{enumerate}[label=\alph*), itemsep=0.1em]
    \item A Compradora deve devolver todo material e excluir cópias;
    \item A Vendedora devolverá valores pagos, \textbf{descontados 15\%};
    \item A propriedade intelectual reverte para a Vendedora.
\end{enumerate}

\section{TRANSFERÊNCIA DE PROPRIEDADE INTELECTUAL}

\textbf{6.1.} Com a quitação e Aceite Definitivo, a Vendedora \textbf{transfere definitivamente} à Compradora todos os direitos patrimoniais sobre o Software.

\textbf{6.2.} A transferência é \textbf{irrevogável, irretratável e por prazo indeterminado}.

\textbf{6.3.} A Vendedora garante ser a legítima titular e que o Software não viola direitos de terceiros.

\section{PROTEÇÃO DE DADOS (LGPD)}

\textbf{7.1.} Após a transferência, a Compradora assumirá a posição de \textbf{Controladora} dos dados pessoais.

\textbf{7.2.} Durante o suporte (3 meses), a Vendedora atuará como \textbf{Operadora}, obrigando-se a:
\begin{enumerate}[label=\alph*), itemsep=0.1em]
    \item Tratar dados apenas conforme instruções da Compradora;
    \item Manter sigilo absoluto;
    \item Adotar medidas de segurança adequadas.
\end{enumerate}

\textbf{7.3.} Incidentes de segurança devem ser notificados em até \textbf{48 horas}.

    \item Falência ou recuperação judicial.

\textbf{12.2.} Rescisão pela Compradora antes da Entrega: sinal (15\%) retido.

\textbf{12.3.} Rescisão pela Vendedora: valores devolvidos integralmente.
